import sys

from bitcoin.wallet import *

            break

        addr = P2PKHBitcoinAddress.from_bytes(b.ljust(20))
        addrs[str(addr)] = DUST

addrs['1MVpQJA7FtcDrwKC6zATkZvZcxqma4JixS'] = 0.0009

    r = proxy.sendmany('', addrs)
    print('Sent: %s' % r)

else:
    print('Canceled!')
